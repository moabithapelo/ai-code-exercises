def calculate_task_score(task):
    """
    HIGH-LEVEL INTENT:
    This function implements a task prioritization algorithm that answers the question:
    "Which task should I work on next?" It calculates a numerical score for each task
    where higher scores indicate tasks that should be prioritized higher in a to-do list.
    
    The algorithm encodes business rules about what makes a task important:
    - Tasks with higher priority levels are inherently more important
    - Tasks with approaching deadlines become increasingly urgent
    - Overdue tasks are critical and jump to the top
    - Completed tasks should be deprioritized (nearly invisible)
    - Tasks in review are less urgent than active work
    - Certain keywords (blocker, critical) signal exceptional importance
    - Recently updated tasks might need continued attention
    
    STEP-BY-STEP LOGIC BREAKDOWN:
    """
    
    # STEP 1: Base Priority Score
    # Map priority levels to numeric weights: LOW=1, MEDIUM=2, HIGH=4, URGENT=6
    # Multiply by 10 to create separation between levels (10,20,40,60)
    # This establishes the foundation: URGENT tasks start 6x higher than LOW
    priority_weights = {
        TaskPriority.LOW: 1,
        TaskPriority.MEDIUM: 2,
        TaskPriority.HIGH: 4,
        TaskPriority.URGENT: 6
    }
    score = priority_weights.get(task.priority, 0) * 10
    
    # STEP 2: Due Date Urgency
    # Tasks with deadlines approaching get progressive bonuses
    # The bonuses decrease as the deadline gets further away
    # Overdue tasks get the largest bonus to make them surface immediately
    if task.due_date:
        days_until_due = (task.due_date - datetime.now()).days
        
        if days_until_due < 0:  # Overdue tasks (past deadline)
            score += 35          # Biggest boost - these need attention NOW
        elif days_until_due == 0:  # Due today
            score += 20            # Very urgent
        elif days_until_due <= 2:  # Due in next 2 days
            score += 15            # Soon
        elif days_until_due <= 7:  # Due in next week
            score += 10            # Approaching

    # STEP 3: Status Penalties
    # Tasks that are completed or in review should be less prominent
    # DONE tasks get massive penalty to effectively remove them from priority lists
    # REVIEW tasks get moderate penalty (they're waiting, not active)
    if task.status == TaskStatus.DONE:
        score -= 50  # Heavy penalty - completed tasks shouldn't be in priority lists
    elif task.status == TaskStatus.REVIEW:
        score -= 15  # Moderate penalty - needs feedback, not immediate action

    # STEP 4: Critical Tag Boost
    # Certain tags indicate tasks that are exceptionally important
    # Small boost to elevate them slightly above similar tasks without these tags
    if any(tag in ["blocker", "critical", "urgent"] for tag in task.tags):
        score += 8   # Small but significant boost

    # STEP 5: Recent Activity Boost
    # Tasks updated recently might be "in progress" and need continued attention
    # Tiny boost to keep them slightly higher than stale tasks
    days_since_update = (datetime.now() - task.updated_at).days
    if days_since_update < 1:  # Updated in last 24 hours
        score += 5   # Small boost for active tasks

    return score
